```python
# 创建测试脚本 test_response_time.py
import requests
import time

test_inputs = [
    "墙体厚度大于200mm",
    "门的高度在2000到2400之间",
    # ... 添加更多测试样本
]

results = []
for input_text in test_inputs:
    times = []
    for _ in range(3):
        start = time.time()
        response = requests.post("http://localhost:8000/api/generate", 
                                json={"text": input_text})
        end = time.time()
        times.append((end - start) * 1000)  # 转换为ms
    results.append({
        "input": input_text,
        "times": times,
        "average": sum(times) / len(times)
    })

# 保存结果到 response_time_results.json
```
